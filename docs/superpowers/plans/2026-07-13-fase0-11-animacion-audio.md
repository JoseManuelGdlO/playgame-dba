# Fase 0 / Plan 11: Animación y Audio Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add motion (screen cross-fades, staggered paper entrance + hover, a progressive-reveal scoring sequence) and a fully code-generated audio layer (SFX + a synthwave ambient loop, two independent mute toggles) on top of the already-built Menu/Hub/Consola screens — no gameplay logic, no Tauri commands, no Rust changes.

**Architecture:** A new standalone ES module `app/src/audio.js` (Web Audio API, zero external audio files) exposes pure SFX-trigger functions plus ambient-loop/mute controls; `app/src/main.js` wires those triggers into existing action points and rewrites three functions (`mostrarPantalla`, `renderBandeja`/`renderPerks`, `mostrarScoring`) to add motion; `app/src/styles.css` gains the supporting transitions/keyframes; `app/src/index.html` gains two small mute buttons.

**Tech Stack:** Vanilla JS (ES modules, already `type="module"` in `index.html`), CSS animations/transitions, Web Audio API. No new dependencies.

## Global Constraints

- All audio is 100% generated by code via the Web Audio API — no external audio files, no real music tracks (licensing risk explicitly rejected by the user during brainstorming).
- The `AudioContext` and the ambient loop must not start until the first user gesture (browser/webview autoplay policy) — bootstrapped from the first click anywhere in the app.
- Music mute and effects mute are two independent toggles, each starting "on" every session — no persistence across app restarts.
- Screen transitions are a simple cross-fade (~250ms), not a slide or anything more elaborate.
- Strictly frontend: only `app/src/*.{js,html,css}` change. No Rust, no new Tauri commands.
- No frontend test runner exists in this project (same as Plans 6-10) — correctness comes from careful diff self-review plus a final manual verification pass in the real running app. This plan's manual verification must include **listening**, not just looking.

- [ ] **Step 1: Create the audio engine module**

Create `app/src/audio.js` with this exact content:

```js
let contexto = null;
let musicaSilenciada = false;
let efectosSilenciados = false;
let ambienteIniciado = false;
let busAmbiente = null;
let indicePatronAmbiente = 0;

const PATRON_AMBIENTE_HZ = [130.81, 164.81, 196.0, 164.81];
const DURACION_NOTA_AMBIENTE_MS = 700;
const FRECUENCIA_PAD_HZ = 65.41;

function obtenerContexto() {
  if (!contexto) {
    contexto = new (window.AudioContext || window.webkitAudioContext)();
  }
  if (contexto.state === "suspended") {
    contexto.resume();
  }
  return contexto;
}

function obtenerBusAmbiente() {
  const ctx = obtenerContexto();
  if (!busAmbiente) {
    busAmbiente = ctx.createGain();
    busAmbiente.gain.value = musicaSilenciada ? 0 : 1;
    busAmbiente.connect(ctx.destination);
  }
  return busAmbiente;
}

function tono(frecuenciaHz, duracionMs, tipo, volumen) {
  if (efectosSilenciados) return;
  const ctx = obtenerContexto();
  const osc = ctx.createOscillator();
  const ganancia = ctx.createGain();
  osc.type = tipo;
  osc.frequency.value = frecuenciaHz;
  ganancia.gain.setValueAtTime(volumen, ctx.currentTime);
  ganancia.gain.exponentialRampToValueAtTime(0.0001, ctx.currentTime + duracionMs / 1000);
  osc.connect(ganancia);
  ganancia.connect(ctx.destination);
  osc.start();
  osc.stop(ctx.currentTime + duracionMs / 1000);
}

function secuenciaTonos(notasHz, duracionNotaMs, tipo, volumen, separacionMs) {
  notasHz.forEach((frecuenciaHz, indice) => {
    setTimeout(() => tono(frecuenciaHz, duracionNotaMs, tipo, volumen), indice * separacionMs);
  });
}

export function sfxClick() {
  tono(600, 60, "square", 0.05);
}

export function sfxTecleo() {
  const variacion = 0.9 + Math.random() * 0.2;
  tono(1200 * variacion, 30, "square", 0.04);
}

export function sfxTick() {
  tono(880, 80, "sine", 0.08);
}

export function sfxExito() {
  secuenciaTonos([523.25, 659.25, 783.99], 150, "triangle", 0.12, 90);
}

export function sfxError() {
  tono(160, 350, "sawtooth", 0.12);
}

export function sfxCierreDia() {
  secuenciaTonos([300, 220], 220, "sine", 0.08, 100);
}

export function sfxAscenso() {
  secuenciaTonos([523.25, 659.25, 783.99, 1046.5], 200, "triangle", 0.14, 110);
}

function reproducirNotaAmbiente(frecuenciaHz) {
  const ctx = obtenerContexto();
  const bus = obtenerBusAmbiente();
  const osc = ctx.createOscillator();
  const ganancia = ctx.createGain();
  osc.type = "triangle";
  osc.frequency.value = frecuenciaHz;
  ganancia.gain.setValueAtTime(0.0001, ctx.currentTime);
  ganancia.gain.exponentialRampToValueAtTime(0.05, ctx.currentTime + 0.05);
  ganancia.gain.exponentialRampToValueAtTime(0.0001, ctx.currentTime + DURACION_NOTA_AMBIENTE_MS / 1000);
  osc.connect(ganancia);
  ganancia.connect(bus);
  osc.start();
  osc.stop(ctx.currentTime + DURACION_NOTA_AMBIENTE_MS / 1000);
}

function reproducirPadAmbiente(frecuenciaHz, duracionMs) {
  const ctx = obtenerContexto();
  const bus = obtenerBusAmbiente();
  const osc = ctx.createOscillator();
  const ganancia = ctx.createGain();
  osc.type = "sine";
  osc.frequency.value = frecuenciaHz;
  ganancia.gain.setValueAtTime(0.0001, ctx.currentTime);
  ganancia.gain.linearRampToValueAtTime(0.03, ctx.currentTime + 0.4);
  ganancia.gain.linearRampToValueAtTime(0.0001, ctx.currentTime + duracionMs / 1000);
  osc.connect(ganancia);
  ganancia.connect(bus);
  osc.start();
  osc.stop(ctx.currentTime + duracionMs / 1000);
}

function agendarSiguienteNotaAmbiente() {
  if (indicePatronAmbiente % PATRON_AMBIENTE_HZ.length === 0) {
    reproducirPadAmbiente(FRECUENCIA_PAD_HZ, PATRON_AMBIENTE_HZ.length * DURACION_NOTA_AMBIENTE_MS);
  }
  reproducirNotaAmbiente(PATRON_AMBIENTE_HZ[indicePatronAmbiente % PATRON_AMBIENTE_HZ.length]);
  indicePatronAmbiente += 1;
  setTimeout(agendarSiguienteNotaAmbiente, DURACION_NOTA_AMBIENTE_MS);
}

export function iniciarAmbiente() {
  if (ambienteIniciado) return;
  ambienteIniciado = true;
  obtenerBusAmbiente();
  agendarSiguienteNotaAmbiente();
}

export function alternarMusica() {
  musicaSilenciada = !musicaSilenciada;
  if (busAmbiente) {
    busAmbiente.gain.value = musicaSilenciada ? 0 : 1;
  }
  return !musicaSilenciada;
}

export function alternarEfectos() {
  efectosSilenciados = !efectosSilenciados;
  return !efectosSilenciados;
}
```

Note on the mute design: `alternarMusica`/`alternarEfectos` never touch `ambienteIniciado` or the scheduler — muting only zeroes `busAmbiente`'s gain, so `agendarSiguienteNotaAmbiente`'s `setTimeout` chain keeps running silently and unmuting is instant (no restart, no gap).

- [ ] **Step 2: Rewrite `mostrarPantalla`**

In `app/src/main.js`, replace the current `mostrarPantalla` function (lines 20-25):

```js
function mostrarPantalla(nombre) {
  pantallaMenu.classList.toggle("oculto", nombre !== "menu");
  appShell.classList.toggle("oculto", nombre === "menu");
  pantallaHub.classList.toggle("oculto", nombre !== "hub");
  pantallaConsola.classList.toggle("oculto", nombre !== "consola");
}
```

with:

```js
const DURACION_TRANSICION_MS = 250;

function alternarPantalla(el, mostrar) {
  if (mostrar) {
    el.classList.remove("oculto");
    void el.offsetHeight;
    el.classList.remove("fade-out");
  } else {
    el.classList.add("fade-out");
    setTimeout(() => {
      if (el.classList.contains("fade-out")) {
        el.classList.add("oculto");
      }
    }, DURACION_TRANSICION_MS);
  }
}

function mostrarPantalla(nombre) {
  alternarPantalla(pantallaMenu, nombre === "menu");
  alternarPantalla(appShell, nombre !== "menu");
  alternarPantalla(pantallaHub, nombre === "hub");
  alternarPantalla(pantallaConsola, nombre === "consola");
}
```

The `void el.offsetHeight;` forces a layout reflow between removing `"oculto"` (which restores `display`) and removing `"fade-out"` (which triggers the opacity transition) — without it, the browser could coalesce both class changes into a single frame and skip the fade-in entirely. The `if (el.classList.contains("fade-out"))` guard in the `setTimeout` prevents a fast toggle-back-and-forth (hide then show within 250ms) from incorrectly re-hiding an element that was already shown again in the meantime.

async function mostrarScoring(score) {
  const tituloEl = document.querySelector("#scoring-titulo");
  const mentorEl = document.querySelector("#scoring-mentor");

  btnCerrarScoring.disabled = true;
  tituloEl.textContent = "";
  tituloEl.className = "";
  mentorEl.textContent = "";
  scoringAscenso.textContent = "";

  const lineas = [
    { span: document.querySelector("#scoring-correctitud"), valor: score.puntaje_correctitud, decimales: 0 },
    { span: document.querySelector("#scoring-velocidad"), valor: score.puntaje_velocidad, decimales: 0 },
    { span: document.querySelector("#scoring-practicas"), valor: score.puntaje_practicas, decimales: 0 },
    { span: document.querySelector("#scoring-dinero"), valor: score.dinero_ganado, decimales: 0 },
    { span: document.querySelector("#scoring-reputacion"), valor: score.reputacion_ganada, decimales: 1 },
  ].map((linea) => ({ ...linea, fila: linea.span.closest("p") }));

  for (const linea of lineas) {

  }

  }
